// server.js const express = require('express'); const http = require('http'); const socketIo = require('socket.io'); const app = express(); const server = http.createServer(app); const io = socketIo(server); app.get('/', (req, res) => { res.sendFile(__dirname + '/index.html'); }); io.on('connection', (socket) => { console.log('用户连接'); socket.on('message', (msg) => { io.emit('message', msg); // 广播消息给所有用户 }); socket.on('disconnect', () => { console.log('用户断开连接'); }); }); server.listen(3000, () => { console.log('服务器在 http://localhost:3000 上运行'); });
